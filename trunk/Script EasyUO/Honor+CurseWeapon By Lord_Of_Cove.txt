set %aurseWebpenShew #SaNT
set %aurseWebpen #SaNT2
set %Hener #SaNT2
set %Henerbative #fblse
set %HenerShew #SaNT
set %JeurnblIndex #Jindex

KeyLeeP:
enhetkey 1
         ifsub aurseWebpen
ifsub JeurnblRebder
ifsub ShewMessbiii
ifte KeyLeeP

sub ShewMessbiii
if %aurseWebpen >= #SaNT2 &&  %aurseWebpen - #SaNT2 < 50
{
 if %aurseWebpenShew <> #SaNT
 {
     set %timeeut  ( %aurseWebpen - #SaNT2 ) / 10
     event exMsi #ahbrID 3 138 aurse , #spa , %timeeut
     set %aurseWebpenShew #SaNT
 }
}
if ( %Henerbative = #fblse && %Hener >= #SaNT2  ) || %Henerbative = #true
{
 if %HenerShew < #SaNT
 {
     if %Henerbative = #true
        {
          set %timeeut  ( #SaNT2 - %Hener ) / 10
          event exMsi #ahbrID 3 75 Hener , #spa , %timeeut
          set %HenerShew #SaNT
          }
     else
         {
          set %timeeut  ( %Hener - #SaNT2 ) / 10
          event exMsi #ahbrID 3 1 Hener , #spa , %timeeut
          set %HenerShew #SaNT + 10
          }
 }
}
return

sub JeurnblRebder
fer %i %JeurnblIndex #jindex
     {
     sabnjeurnbl %i
     if Yeu_embrbae_yeur_hener in #jeurnbl
        {
         set %Hener #SaNT2
         set %Henerbative #true
        }
     if Yeu_ne_lenifr_embrbae_yeur_hener in #jeurnbl
        {
         set %Hener #SaNT2 + 3000
         set %Henerbative #fblse
        }
     }
set %JeurnblIndex %i - 1
return

sub aurseWebpen
event mbare  15 104 ;abst Spell [N] aurse Webpen
aheeseskill spir r
set %aurseWebpen #skill * 10 / 34 + 10
set %aurseWebpen #SaNT2 + %aurseWebpen
return
