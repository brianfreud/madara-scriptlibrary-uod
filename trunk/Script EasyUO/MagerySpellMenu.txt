  menu HideEUO
	menu Clear
	menu Window Title Druethelan's EUO Magic Tool
	menu Window Color Silver
	menu Window Size 277 225
	menu Font Transparent #true
	menu Font Align Right
	menu Shape EUOShape1 0 88 277 5 3 7 1 Black 7 WindowFrame
	menu Font Name MS Sans Serif
	menu Font Size 8
	menu Font Style
	menu Font Color WindowText
	menu Button Circle1 6 4 64 35 1st
	menu Button Circle2 74 4 64 35 2nd
	menu Button Circle3 142 4 64 35 3rd
	menu Button Circle4 210 4 64 35 4th
	menu Button Circle5 6 44 64 35 5th
	menu Button Circle6 74 44 64 35 6th
	menu Button Circle7 142 44 64 35 7th
	menu Button Circle8 210 44 64 35 8th
	menu Button nil 38 125 200 35 Select a circle to begin!
  menu Button Med 94 180 88 35 Meditate
	menu Show 421 270
	set #menubutton nil

MainLoop:
if #menubutton <> nil
  {
      if #menubutton = CLOSED
        halt
      if #menubutton = Med
     {
        gosub Med
     }
      if circle in #menubutton
     {
        menu delete nil
        gosub #menubutton
     }
     else
     {
        gosub castSpell #menubutton
     }
     set #menubutton nil
  }
  wait 5
  goto MainLoop
