useobject righthand pause 500 // Uses item, specificaly a pickaxe
 if weight >= 500 //checks packs weight before smelting
  msg 'Blargh! So Heavy!' //An alarm basically to let you know your character needs a minute to smelt
  pause 1500
  for 2 // Loops up until "endfor" giving this section a chance to smelt both regular ore and colored ore
   pause 750
   usetype 0x19b9 //Ore Type ID
   pause 500
   target 0x41639543 //Port'o'forge ID
   pause 500
  endfor
 endif
