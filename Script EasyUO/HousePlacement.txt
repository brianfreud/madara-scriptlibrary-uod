int

Set %Xcoord 800
Set %Ycoord 555

event SysMessage Welcome to TL's Meticulous House Placer
event macro 19 0
wait 5
event macro 8 1
wait 10
contpos 850 0
wait 5
event macro 8 2
wait 5
contpos 850 300
event macro 8 7
wait 10
contpos 850 500
wait 5

finditem CWH
set %house #findID
fishingLoop2:
set #lObjectID %house
event macro 17 0
wait1:
wait 5
if  #CONTSIZE = 270_145
{
click 73 122
wait 5
goto placeloop
}
goto wait1

placeloop:
wait 5
if  #CONTSIZE = 520_420
{
click 73 128
goto housetarget
}
goto wait 2

housetarget:
target 2s
if %xcoord =< 14
{
set %ycoord %ycoord - 15
Set %xcoord 755
}
if %ycoord =< 14
{
event SysMessage House Failed
event SysMessage At location %Xcoord %YCoord
event SysMessage Placement Failed!
wait 60s
halt
}
set %xcoord %xcoord - 15
wait 5
click %Xcoord %YCoord
wait 1s
if  #CONTSIZE = 420_280
{
placed:
event SysMessage House Placed!
event SysMessage At location %Xcoord %YCoord
wait 10s
goto placed
halt
}
event SysMessage House Failed
event SysMessage At location %Xcoord %YCoord
wait 10
goto placeloop
