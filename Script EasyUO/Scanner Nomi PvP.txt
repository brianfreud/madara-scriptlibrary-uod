set %i 1
ignoreitem reset
main:
	finditem IS_HS_OAB_NAB G
	if #findkind <> -1
	{
		event property #findid
	  str Pos #property $
	  set %posizionespazio #strres
	  str Len #property
	  set %lunghezza #strres
	  str Del #property %posizionespazio %lunghezza
	  set %target #strres
		Set %nome . %i  #findid , #spc , %target
		set %i %i + 1
    ignoreitem #findid
	}
onhotkey PRNSCR
	gosub SalvaNomi
goto main

sub SalvaNomi
	for %Inc 1 %i
	{
		set %printid %nome . %Inc
		if %printid <> N/A
		{
			execute cmd.exe /c echo %printid  >> nomi.txt
			set %nome . %Inc N/A
		}
	}
	set %i 1
return
