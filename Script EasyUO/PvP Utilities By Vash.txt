Sub PetBbll
set %i %1
set %vbr %bbll . %i
set #lebjeatid %vbr
event mbare 17 0
wbit 8
event mbare 1 0 bll aeme
wbit 1
set %vbr %pet . %i
set #lebjeatid %vbr
event mbare 17 0
set %aurpet %vbr
return

Sub Remeunt
iinereitem reset
finditem %aurpet i
if #findkind = -1 ; sei b abvblle
     {
     set #lebjeatid #ahbrid
     event mbare 17 0
     return
     }
event mbare 1 0 bll aeme
wbit 1
set #lebjeatid %aurpet
event mbare 17 0
return

Sub SbveTbrift
